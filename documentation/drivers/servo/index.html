Servo

Driver

Provides an interface for making servos easy to work and interact with from Cylon.js.

How to establish a connection and setup the driver

JavaScript

var Cylon = require('cylon');

Cylon.robot({
  connection: {name: 'arduino', adaptor: 'firmata', port: '/dev/ttyACM0'},

  device: {name: 'servo', driver: 'servo', pin: 3},
...

Cylon = require 'cylon'

Cylon.robot
  connection:
    name: 'arduino', adaptor: 'firmata', port: '/dev/ttyACM0'

  device:
    name: 'servo', driver: 'servo', pin: 3
...

Commands:

angle → true | nil
Moves the servo to the specified angle, angle must be an integer value between 0 and 180.
currentAngle → integer_value (0 - 180)
Returns the current angle of the servo, an integer value between 0 and 180.

Events

start
Sent when the device has been started and is ready to use.

How to use

This example moves a servo from angle 45 to 90 to 135.

JavaScript

var Cylon = require('cylon');

Cylon.robot({
  connection: {name: 'arduino', adaptor: 'firmata', port: '/dev/ttyACM0'},

  device: {name: 'servo', driver: 'servo', pin: 3},

  work: function(my) {
    var angle = 45 ;
    my.servo.angle(angle);
    every((1).second(), function() {
      angle = angle + 45 ;
      if (angle > 135) {
        angle = 45
      }
      my.servo.angle(angle);
    });
  }
}).start();

CoffeeScript

Cylon = require 'cylon'

Cylon.robot
  connection:
    name: 'arduino', adaptor: 'firmata', port: '/dev/ttyACM0'

  device:
    name: 'servo', driver: 'servo', pin: 3

  work: (my) ->
    angle = 45
    my.servo.angle angle
    every 1.second(), ->
      angle += 45
      angle = 45 if angle > 135
      my.servo.angle angle

.start()
