Force

Driver

This module provides an adaptor and drivers for Force.com .It uses the nforce module (https://github.com/kevinohara80/nforce) created by @kevinohara80 thank you!

For more info about the Force platform go to http://www.force.com/.

How to establish a connection and setup the driver

JavaScript

Cylon = require('cylon');

Cylon.robot({
  connection: {
    name: 'sfcon',
    adaptor: 'force'
  },
  device: {
    name: 'salesforce',
    driver: 'force'
  }
  ...

CoffeeScript

  Cylon = require('cylon')

  Cylon.robot
    connection:
      name: 'sfcon', adaptor: 'force'

    device:
      name: 'salesforce', driver: 'force'
   ...

Commands:

authenticate
This allows you to authenticate the device.
subscribe
This allows you to suscribe the device.
push
This allows you to push data.

Events:

connect
Gets triggered when it´s connected.
authenticate
Gets triggered when it´s authenticated.
subscribe
Gets triggered when it´s subscribed.
stop
Stop the driver.
start
Sent when the device has been started and is ready to use.

How to use

This example uses the sphero with cylonjs-force

JavaScript

Cylon = require('cylon');

Cylon.robot({
  connection: {
    name: 'sfcon',
    adaptor: 'force',
    sfuser: process.env.SF_USERNAME,
    sfpass: process.env.SF_SECURITY_TOKEN,
    orgCreds: {
      clientId: process.env.SF_CLIENT_ID,
      clientSecret: process.env.SF_CLIENT_SECRET,
      redirectUri: 'http://localhost:3000/oauth/_callback'
    }
  },
  device: {
    name: 'salesforce',
    driver: 'force'
  },
  work: function(me) {
    var toSend;
    me.salesforce.on('start', function() {
      return me.salesforce.subscribe('/topic/SpheroMsgOutbound', function(data) {
        return Logger.info("Sphero: " + data.sobject.Sphero_Name__c + ", data Content: " + data.sobject.Content__c + ", SM_Id: " + data.sobject.Id);
      });
    });
    toSend = "{ \"identifier\" :\"" + me.name + "\", \"msg\": \"" + 'Salesforce Bot #2' + "\" }";
    return me.salesforce.push('SpheroController', 'POST', toSend);
  }
}).start();

CoffeeScript

Cylon = require('cylon')

Cylon.robot
  connection:
    name: 'sfcon',
    adaptor: 'force',
    sfuser: process.env.SF_USERNAME,
    sfpass: process.env.SF_SECURITY_TOKEN,
    orgCreds: {
      clientId: process.env.SF_CLIENT_ID,
      clientSecret: process.env.SF_CLIENT_SECRET,
      redirectUri: 'http://localhost:3000/oauth/_callback'
    }

  device:
    name: 'salesforce', driver: 'force'

  work: (me) ->

    me.salesforce.on('start', () ->
      me.salesforce.subscribe('/topic/SpheroMsgOutbound', (data) ->
        Logger.info "Sphero: #{ data.sobject.Sphero_Name__c }, data Content: #{ data.sobject.Content__c }, SM_Id: #{ data.sobject.Id }"
      )
    )

    toSend = "{ \"identifier\" :\"#{ me.name }\", \"msg\": \"#{ 'Salesforce Bot #2' }\" }"
    me.salesforce.push('SpheroController', 'POST', toSend)
.start()
